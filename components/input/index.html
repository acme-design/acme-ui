位置聚焦

聚焦额外配置属性

复制
  1() => {
  2  const inputRef = React.createRef();
  3  const jumpToStart = () => {
  4    inputRef.selectionStart = 0;
  5    inputRef.current.focus();
  6    inputRef.current.setSelectionRange(0, 0);
  7  };
  8  const jumpToEnd = () => {
  9    const valueLen = inputRef.current.value.length;
 10    inputRef.current.focus();
 11    inputRef.current.setSelectionRange(valueLen, valueLen);
 12  };
 13  const selectAll = () => {
 14    inputRef.current.select();
 15  };
 16  return (
 17    <div>
 18      <Input
 19        ref={inputRef}
 20        placeholder="请输入内容"
 21        style={{ width: "400px" }}
 22        value="这是一个输入框"
 23      />
 24      <div
 25        style={{
 26          width: "400px",
 27          marginTop: "24px",
 28          display: "flex",
 29          justifyContent: "space-between",
 30        }}
 31      >
 32        <Button mode="outlined" size="small" onClick={jumpToStart}>
 33          开始位置
 34        </Button>
 35        <Button mode="outlined" size="small" onClick={selectAll}>
 36          聚焦并全选
 37        </Button>
 38        <Button mode="outlined" size="small" onClick={jumpToEnd}>
 39          结束位置
 40        </Button>
 41      </div>
 42    </div>
 43  );
 44};
 45
>

API

参数(*为必填)说明类型默认值
className样式类名string
modeinput 类型 TODO 暂时没有样式变化 "contained" | "outlined" | "minimal"
size输入框大小 "default" | "small" | "large"default
value输入框值string | number | readonly string[]
defaultValue输入框默认值 非受控string | number | readonly string[]
placeholder输入框内容提示string
disabled禁用状态boolean
status状态 "loading" | "error" | "success"
fullWidth是否撑满父元素boolean
startElementinput内部最前面的元素ReactNode
endElementinput内部最后面的元素ReactNode
onChange输入框内容变化时的回调((e?: ChangeEvent<HTMLInputElement> | undefined) => void)
onClear输入框清空的回调(() => void)
limit输入框字数限制number
clear是否展示清空按钮boolean
onFocus输入框获得焦点的回调((e: FocusEvent<HTMLInputElement>) => void)
onBlur输入框失去焦点的回调((e: FocusEvent<HTMLInputElement>) => void)
type输入框类型string text
style输入框最外层样式CSSProperties
Form Group表单组Checkbox复选框
