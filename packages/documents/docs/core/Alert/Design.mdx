import {Playground, Props} from "~controls";
import { Alert } from "@acme-ui/core";

## 演示
该演示使您可以预览下拉菜单的不同类型，以及不同类型的不同状态。
<Playground
    links={{
        codeSandbox: '',
        codePen: '',
        stackblitz: '',
    }}
    defaultComProps={{type: 'info'}}
    headWrapper={Playground.renderChecker([
        {label: '默认', onClick: (p) => ({...p, type: 'info'}),highlight: (p) => (p.type === 'info') },
        {label: '警告', onClick: (p) => ({...p, type: 'warning'}),highlight: (p) => (p.type === 'warning')},
        {label: '成功', onClick: (p) => ({...p, type: 'success'}),highlight: (p) => (p.type === 'success')},
        {label: '失败', onClick: (p) => ({...p, type: 'error'}),highlight: (p) => (p.type === 'error')},
    ])}
>
    <Alert data-mark>消息提示文案</Alert>
</Playground>

## 基础用法
基本使用方法，有四种样式可以选择info、success、warning、error。
<Playground >
    <Alert>消息提示文案</Alert>
    <br/>
    <Alert type="success">成功提示文案</Alert>
    <br/>
    <Alert type="warning">警告提示文案</Alert>
    <br/>
    <Alert type="error">错误提示文案</Alert>
</Playground>
