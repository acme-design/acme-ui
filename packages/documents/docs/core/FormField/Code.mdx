## 配合 Radio 使用

<Playground>
  {() => {
    const [sex, setSex] = React.useState('1');
    const onSexChange = (e) => {
      const value = e.target.value;
      setSex(value);
    };
    return (
      <div>
        <FormField
          style={{ marginBottom: 24 }}
          id="demo-radio"
          label="正常的性别"
          hint="请选择选项"
        >
          <Radio.Group layout="horizontal" value={sex} onChange={onSexChange}>
            <Radio label="男" value="1" style={{ marginRight: 32 }} />
            <Radio label="女" value="0" />
          </Radio.Group>
        </FormField>
        <FormField
          style={{ marginBottom: 24 }}
          id="demo-error-radio"
          label="错误的性别"
          hint="该选项错误"
          status="error"
        >
          <Radio.Group value={sex} onChange={onSexChange}>
            <Radio label="男" value="1" style={{ marginBottom: 8 }} />
            <Radio label="女" value="0" />
          </Radio.Group>
        </FormField>
        <FormField
          style={{ marginBottom: 24 }}
          id="demo-disabled-radio"
          label="禁用的性别"
          hint="该选项禁止选择🚫"
          disabled
        >
          <Radio.Group value={sex} onChange={onSexChange}>
            <Radio label="男" value="1" style={{ marginBottom: 8 }} />
            <Radio label="女" value="0" />
          </Radio.Group>
        </FormField>
      </div>
    );
  }}
</Playground>
